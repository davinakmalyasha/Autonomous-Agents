import itertools
import time


class PriorityQueueWithAging:
    """
    A priority queue that prevents starvation by boosting the effective priority
    of waiting items over time (aging).

    Effective priority = base_priority + aging_factor * (current_time - enqueue_time)
    Higher effective priority is retrieved first.

    The current implementation uses a linear scan for get() (O(n)) to always
    compute the true effective priority. This is simple, correct, and sufficient
    for moderate queue sizes. A heap-based version with lazy updates could be
    used for better scalability.

    Parameters
    ----------
    aging_factor : float
        How much effective priority increases per second of waiting.
        Larger values cause older items to overtake newer items faster.
    time_func : callable, optional
        Function that returns the current time in seconds.
        Defaults to time.time.  Can be a fake clock in tests.
    """

    def __init__(self, aging_factor=1.0, time_func=None):
        self.aging_factor = aging_factor
        self._time_func = time_func or time.time
        self._items = []  # list of (item, base_priority, enqueue_time, counter)
        self._counter = itertools.count()

    # ------------------------------------------------------------------
    # Public interface
    # ------------------------------------------------------------------

    def put(self, item, priority=0):
        """Insert an item with the given base priority (higher = more important)."""
        entry = (item, priority, self._time_func(), next(self._counter))
        self._items.append(entry)

    def get(self, self):
        """
        Remove and return the item with the highest effective priority.

        Raises IndexError if the queue is empty.
        """
        if not self._items:
            raise IndexError("get from empty queue")

        now = self._time_func()
        best_idx = 0
        best_effective = self._effective(self._items[0], now)

        for i in range(1, len(self._items)):
            eff = self._effective(self._items[i], now)
            if eff > best_effective:
                best_effective = eff
                best_idx = i

        # Swap with last element for O(1) removal
        last_idx = len(self._items) - 1
        self._items[best_idx], self._items[last_idx] = (
            self._items[last_idx],
            self._items[best_idx],
        )
        item, _priority, _enqueued, _counter = self._items.pop()
        return item

    def qsize(self, self):
        """Return the number of items currently in the queue."""
        return len(self._items)

    def empty(self, self):
        """Return True if the queue is empty, False otherwise."""
        return len(self._items) == 0

    # ------------------------------------------------------------------
    # Internal helpers
    # ------------------------------------------------------------------

    def _effective(self, entry, now):
        """Compute effective priority of one entry at the given time."""
        _item, base_priority, enqueue_time, _counter = entry
        age = now - enqueue_time
        return base_priority + self.aging_factor * age
