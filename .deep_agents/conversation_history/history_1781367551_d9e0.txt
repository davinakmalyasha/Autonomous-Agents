from router import Router, HTTPException


def handler_home(params):
    return {"message": "Home", "params": params}


def handler_users(params):
    return {"message": "Users list", "params": params}


def handler_user(params):
    return {"message": "User detail", "params": params}


def handler_create(params):
    return {"message": "Create user", "params": params}


def handler_edit(params):
    return {"message": "Edit user", "params": params}


def handler_delete(params):
    return {"message": "Delete user", "params": params}


def handler_items(params):
    return {"message": "Items list", "params": params}


def handler_item(params):
    return {"message": "Item detail", "params": params}


@pytest.fixture
def router():
    r = Router()
    r.add_route("GET", "/", handler_home)
    r.add_route("GET", "/users", handler_users)
    r.add_route("GET", "/users/{id}", handler_user)
    r.add_route("POST", "/users", handler_create)
    r.add_route("PUT", "/users/{id}", handler_edit)
    r.add_route("DELETE", "/users/{id}", handler_delete)
    r.add_route("GET", "/items", handler_items)
    r.add_route("GET", "/items/{item_id}", handler_item)
    return r
