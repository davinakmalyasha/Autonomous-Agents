import time
from cryptography.fernet import Fernet


class PasswordVault:
    """A credential vault with brute-force backoff protection.

    Uses Fernet symmetric encryption.  On each consecutive failed decryption
    the vault sleeps for an exponentially increasing delay (starting at 0.1 s,
    doubling each failure, capped at 5 s).  A successful decryption resets
    the failure counter.
    """

    def __init__(self, key: bytes | None = None):
        if key is None:
            key = Fernet.generate_key()
        self._fernet = Fernet(key)
        self._failure_count = 0
        self._base_delay = 0.1          # seconds
        self._max_delay = 5.0           # seconds

    @property
    def key(self) -> bytes:
        return self._fernet._encrypting_key

    def encrypt(self, plaintext: str) -> str:
        """Encrypt a string and return a base64 token."""
        return self._fernet.encrypt(plaintext.encode()).decode()

    def decrypt(self, token: str) -> str:
        """Decrypt a token.

        Raises InvalidToken on failure after applying backoff delay.
        On success the failure counter is reset.
        """
        try:
            plaintext = self._fernet.decrypt(token.encode()).decode()
            self._failure_count = 0      # reset on success
            return plaintext
        except Exception as exc:          # InvalidToken, etc.
            self._failure_count += 1
            delay = min(self._base_delay * (2 ** (self._failure_count - 1)),
                        self._max_delay)
            time.sleep(delay)
            raise

    @property
    def failure_count(self) -> int:
        return self._failure_count

    def reset_failures(self) -> None:
        """Manually reset the failure counter."""
        self._failure_count = 0

import time
import pytest
from vault import PasswordVault


class TestPasswordVaultBackoff:
    """Verify the brute-force backoff mechanism."""

    @pytest.fixture
    def vault(self):
        return PasswordVault()

    def test_encrypt_decrypt_success(self, vault):
        """Happy path: encrypt and decrypt a secret."""
        secret = "my_secret_pass"
        token = vault.encrypt(secret)
        assert vault.decrypt(token) == secret
        assert vault.failure_count == 0

    def test_failure_resets_on_success(self, vault):
        """Failure count resets to 0 after a successful decrypt."""
        secret = "keep_this_safe"
        token = vault.encrypt(secret)

        # cause one failure
        with pytest.raises(Exception):
            vault.decrypt("bad_token")
        assert vault.failure_count == 1

        # successful decrypt resets
        assert vault.decrypt(token) == secret
        assert vault.failure_count == 0

    def test_exponential_backoff_timing(self, vault):
        """Consecutive failures produce approximately exponential delays."""
        delays = []
        for _ in range(5):  # 0.1, 0.2, 0.4, 0.8, 1.6 s expected
            start = time.perf_counter()
            with pytest.raises(Exception):
                vault.decrypt("bad")
            elapsed = time.perf_counter() - start
            delays.append(round(elapsed, 2))

        # don't enforce exact numbers, but check they increase
        for i in range(1, len(delays)):
            assert delays[i] > delays[i - 1], (
                f"Delay did not increase: {delays}"
            )

    def test_max_backoff_cap(self, vault):
        """Delay never exceeds 5 seconds."""
        # force many failures to reach cap
        for _ in range(20):
            with pytest.raises(Exception):
                vault.decrypt("junk")

        # one more failure should sleep ~5 s
        start = time.perf_counter()
        with pytest.raises(Exception):
            vault.decrypt("junk")
        elapsed = time.perf_counter() - start
        assert elapsed <= 5.5, f"Delay {elapsed} exceeded max cap"

    def test_reset_after_max_backoff(self, vault):
        """After hitting max backoff, a success resets everything."""
        secret = "reset_test"
        token = vault.encrypt(secret)

        # hammer failures
        for _ in range(20):
            with pytest.raises(Exception):
                vault.decrypt("nope")

        assert vault.failure_count > 0

        # successful decrypt resets
        assert vault.decrypt(token) == secret
        assert vault.failure_count == 0

        # next failure should be back to base delay
        start = time.perf_counter()
        with pytest.raises(Exception):
            vault.decrypt("wrong")
        elapsed = time.perf_counter() - start
        # should be around 0.1 s, allow some tolerance
        assert elapsed < 0.3, f"Backoff not reset, slept {elapsed:.3f}s"
