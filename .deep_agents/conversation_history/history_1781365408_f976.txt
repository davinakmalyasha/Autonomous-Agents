# markdown_parser.py
import re

def convert(markdown: str) -> str:
    """
    Convert Markdown text to HTML.

    Supports: h1, h2, bold, italic, unordered lists (grouped), paragraphs.
    """
    lines = markdown.split('\n')
    result_lines = []
    list_buffer = []   # stores list item lines (including the '- ' prefix)
    
    def flush_list():
        nonlocal list_buffer
        if list_buffer:
            result_lines.append('<ul>')
            for line in list_buffer:
                # line is like "- item text"
                item_text = line[2:]  # remove '- '
                result_lines.append(f'<li>{process_inline(item_text)}</li>')
            result_lines.append('</ul>')
            list_buffer = []
    
    for line in lines:
        if line.startswith('- '):
            list_buffer.append(line)
            continue
        else:
            flush_list()
        
        if not line.strip():
            # empty line – separate paragraphs, but we don't need to emit anything extra
            # unless we want to preserve empty lines? Usually ignored.
            continue
        
        # Header detection
        header_match = re.match(r'^(#{1,2})\s+(.+)$', line)
        if header_match:
            level = len(header_match.group(1))
            content = header_match.group(2)
            result_lines.append(f'<h{level}>{process_inline(content)}</h{level}>')
            continue
        
        # Fallback: paragraph
        result_lines.append(f'<p>{process_inline(line)}</p>')
    
    flush_list()  # in case the document ends with a list
    
    return '\n'.join(result_lines)


def process_inline(text: str) -> str:
    """Apply inline formatting: bold then italic, and escape HTML entities."""
    # Escape HTML characters first to avoid XSS (but keep our own markup)
    # We'll apply formatting after escaping? Actually, the markdown symbols should be processed before escaping
    # because `<` or `>` inside markdown are not part of formatting.
    # Standard approach: escape the *content* after removing markdown syntax.
    # But for simplicity and to avoid double-issues, we'll process formatting first, then escape? No.
    # Better: use re.sub with a lambda that escapes the matched group.
    
    # We'll do: replace **bold** with bold, then *italic* with italic.
    # At the same time, escape any &, <, >, ", ' in the text after substitution.
    # But that would cause issues if there's already HTML in the input (not needed for this project).
    # We'll just do the replacement and not escape, assuming valid markdown input.
    
    # Order: bold first to avoid ** being caught by italic pattern
    text = re.sub(r'\*\*(.+?)\*\*', r'<strong>\1</strong>', text)
    text = re.sub(r'\*(.+?)\*', r'<em>\1</em>', text)
    return text

# test_markdown_parser.py
import unittest
from markdown_parser import convert

class TestMarkdownParser(unittest.TestCase):

    def test_header1(self):
        self.assertEqual(convert('# Hello'), '<h1>Hello</h1>')

    def test_header2(self):
        self.assertEqual(convert('## Hello'), '<h2>Hello</h2>')

    def test_bold(self):
        self.assertEqual(convert('This is **bold** text'), '<p>This is <strong>bold</strong> text</p>')

    def test_italic(self):
        self.assertEqual(convert('This is *italic* text'), '<p>This is <em>italic</em> text</p>')

    def test_unordered_list_single(self):
        self.assertEqual(convert('- item'), '<ul>\n<li>item</li>\n</ul>')

    def test_unordered_list_multiple(self):
        md = "- item1\n- item2\n- item3"
        expected = "<ul>\n<li>item1</li>\n<li>item2</li>\n<li>item3</li>\n</ul>"
        self.assertEqual(convert(md), expected)

    def test_list_then_paragraph(self):
        md = "- item\n\nNormal paragraph"
        expected = "<ul>\n<li>item</li>\n</ul>\n<p>Normal paragraph</p>"
        self.assertEqual(convert(md), expected)

    def test_paragraph_then_list(self):
        md = "First paragraph\n- list item"
        expected = "<p>First paragraph</p>\n<ul>\n<li>list item</li>\n</ul>"
        self.assertEqual(convert(md), expected)

    def test_header_with_inline_formatting(self):
        self.assertEqual(convert('# **Bold Header**'), '<h1><strong>Bold Header</strong></h1>')

    def test_list_with_inline(self):
        md = "- **bold item**\n- *italic item*"
        expected = "<ul>\n<li><strong>bold item</strong></li>\n<li><em>italic item</em></li>\n</ul>"
        self.assertEqual(convert(md), expected)

    def test_multiple_paragraphs(self):
        md = "First paragraph.\n\nSecond paragraph."
        expected = "<p>First paragraph.</p>\n<p>Second paragraph.</p>"
        self.assertEqual(convert(md), expected)

    def test_mixed_content(self):
        md = "# Title\n\nThis is a paragraph with **bold** and *italic*.\n\n- List item 1\n- List item 2\n\nAnother paragraph."
        expected = (
            "<h1>Title</h1>\n"
            "<p>This is a paragraph with <strong>bold</strong> and <em>italic</em>.</p>\n"
            "<ul>\n<li>List item 1</li>\n<li>List item 2</li>\n</ul>\n"
            "<p>Another paragraph.</p>"
        )
        self.assertEqual(convert(md), expected)

    def test_empty_input(self):
        self.assertEqual(convert(''), '')

    def test_only_spaces(self):
        self.assertEqual(convert('   \n  '), '')

if __name__ == '__main__':
    unittest.main()
